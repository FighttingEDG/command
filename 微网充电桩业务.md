## 枪的状态不对，该桩充电中解决
- 查看是不是有正在充电状态的枪：select * FROM ec_charge_order  WHERE gun_id=000000000000451 and `order_status` =2;
- 手动修改该笔订单的充电状态：UPDATE ec_charge_order SET order_status = 1 WHERE gun_id = '000000000000451' AND order_status = 2;

## 当前有多少人在充电，充电的时候会冻结
-SELECT c.id, c.phone, c.total, c.frozen_amount
FROM ec_customer c
WHERE c.frozen_amount > 0
  AND NOT EXISTS (
    SELECT 1 FROM ec_charge_order o 
    WHERE o.create_user_id = c.id 
      AND o.order_status IN ('2', '6')
  )
ORDER BY c.frozen_amount DESC
LIMIT 500;

## 把“订单还在进行中（2/6）”和“枪当前状态（ec_pile）”对照起来，找不一致。
SELECT o.id, o.order_no, o.gun_id, o.order_status, o.update_time,
       p.lx_status, p.status AS pile_status, p.update_time AS pile_update_time
FROM ec_charge_order o
LEFT JOIN ec_pile p ON p.gun_id = o.gun_id
WHERE o.order_status IN ('2','6')
ORDER BY o.update_time ASC;
